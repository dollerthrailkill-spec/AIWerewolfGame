async function loadProviders() {
    try {
        const resp = await fetch("/api/config");
        const data = await resp.json();
        providers = data.providers ? Object.values(data.providers) : [];
        // 同步更新 gameState.providers 为字典格式
        gameState.providers = data.providers ? data.providers : {};
        console.log('[DEBUG] Loaded', providers.length, 'providers from backend');
        updateStartButtonState();
    } catch (e) {
        console.error("Failed to load providers:", e);
        providers = [];
        gameState.providers = {};
        updateStartButtonState();
    }
}
 
function loadFromLocalStorage() {
    // 从本地加载角色配置
    const saved = localStorage.getItem('aiWerewolfRoles');
    if (saved) {
        try {
            const savedRoles = JSON.parse(saved);
            // 合并保存的角色配置（保留默认的 avatar、device、model 等字段）
            for (const key in savedRoles) {
                if (savedRoles[key] && roles[key]) {
                    roles[key] = { ...roles[key], ...savedRoles[key] };
                }
            }
            console.log('[DEBUG] Loaded roles from localStorage');
        } catch (e) {
            console.error('Failed to parse saved roles:', e);
        }
    }
}

// ==================== 消息处理 ====================
function enqueueMessage(data) {
    console.log('[DEBUG] Enqueue message:', data.type, data);
    messageQueue.push(data);
    if (!isProcessingMessage) {
        processNextMessage();
    }
}
 
async function processNextMessage() {
    if (messageQueue.length === 0) {
        isProcessingMessage = false;
        return;
    }
 
    // 手动模式下，如果有 messageResumeResolve，说明消息队列被暂停，等待恢复
    // 但 vote_start、vote_result、night_result 等关键消息需要立即处理
    if (messageResumeResolve) {
        const nextMsg = messageQueue[0];
        const urgentTypes = ['vote_start', 'vote_result', 'night_result', 'phase_change', 'game_over', 'eulogy_start', 'eulogy_end'];
        if (!urgentTypes.includes(nextMsg.type)) {
            console.log('[DEBUG] Message queue paused, waiting for user to click next');
            await new Promise(resolve => {
                const check = () => {
                    if (!messageResumeResolve) {
                        resolve();
                    } else {
                        setTimeout(check, 100);
                    }
                };
                check();
            });
        } else {
            console.log('[DEBUG] Urgent message type:', nextMsg.type, ', processing immediately');
        }
    }
 
    isProcessingMessage = true;
    const data = messageQueue.shift();
    try {
        await handleGameMessage(data);
    } catch (err) {
        console.error('[DEBUG] Error processing message:', data.type, err);
    }
    // 使用 setTimeout 避免递归过深导致调用栈溢出，同时让 UI 有机会更新
    setTimeout(() => processNextMessage(), 0);
}
 
async function handleGameMessage(data) {
    const type = data.type;
    console.log('[DEBUG] Handling message type:', type, data);
 
    switch (type) {
        case "game_start":
            await onGameStart(data);
            break;
        case "phase_change":
            await onPhaseChange(data);
            return;
        case "sub_phase":
            updateCenterText(data.message);
            updateGameStep(data.message);
            break;
        case "night_action_log":
            if (data.actions) {
                await playNightActionAnimations(data.actions);
            }
            return;
        case "witch_info":
            showCenterBanner("女巫得知", `今晚${data.killed_player}被狼人杀害`, "kill-banner", 2000);
            break;
        case "night_result":
            await onNightResult(data);
            return;
        case "dawn":
            updateCenterText("天亮了...");
            updateGameStep('天亮了');
            break;
        case "current_speaker":
            await onCurrentSpeaker(data);
            return;
        case "speech":
            await onSpeech(data);
            return;
        case "vote_start":
            await onVoteStart(data);
            return;
        case "vote_details":
            await onVoteDetails(data);
            return;
        case "vote_result":
            await onVoteResult(data);
            return;
        case "hunter_shoot":
            showCenterBanner("猎人发动技能", data.message, "kill-banner", 2000);
            break;
        case "hunter_result":
            await onHunterResult(data);
            return;
        case "eulogy_start":
            showEulogyOverlay(data.player || '', data.message || '');
            return;
        case "eulogy_speech":
            hideEulogyOverlay();
            await onSpeech({
                player: data.player,
                player_id: data.player_id,
                content: data.content,
                is_eulogy: true
            });
            return;
        case "eulogy_end":
            hideEulogyOverlay();
            return;
        case "ai_thinking":
            break;
        case "ai_reasoning":
            addReasoningMessage(data.player, data.reasoning);
            return;
        case "game_paused":
            gameState.isPaused = true;
            const pauseBtn = document.getElementById('pause-btn');
            if (pauseBtn) {
                pauseBtn.innerHTML = '<i class="fa fa-play"></i> 继续';
            }
            break;
        case "game_resumed":
            gameState.isPaused = false;
            const resumeBtn = document.getElementById('pause-btn');
            if (resumeBtn) {
                resumeBtn.innerHTML = '<i class="fa fa-pause"></i> 暂停';
            }
            break;
        case "game_over":
            await onGameOver(data);
            return;
        case "error":
            console.error('[DEBUG] Error from server:', data.message);
            alert(data.message);
            break;
        default:
            console.warn('[DEBUG] Unknown message type:', type);
    }
}

let currentSpeechTimeout = null;
let currentTypeWriterTimeout = null;
 
function onSpeech(data) {
    return new Promise((resolve) => {
        const player = gameState.players.find(p => p.name === data.player);
        const roleInfo = getRoleInfo(player ? player.role : null);
 
        // 显示发言区域
        const speechArea = document.getElementById('speech-area');
        if (!speechArea) {
            console.error('[DEBUG] Speech area not found!');
            resolve();
            return;
        }
 
        // 如果是非自动播放模式且已经有发言在显示，则将新发言加入队列
        const isCurrentlyVisible = !speechArea.classList.contains('hidden');
 
        if (!gameState.autoPlay && isCurrentlyVisible) {
            console.log('[DEBUG] Queueing speech for manual mode');
            // 将发言加入队列，等待用户点击"下一位"
            if (!gameState.speechQueue) {
                gameState.speechQueue = [];
            }
            gameState.speechQueue.push({ data, resolve });
            console.log('[DEBUG] Speech queued');
            return;
        }
 
        // 显示发言
        speechArea.classList.remove('hidden');
        console.log('[DEBUG] Speech area now visible');
 
        const speakerAvatar = document.getElementById('speaker-avatar');
        const speakerName = document.getElementById('speaker-name');
        const speakerRole = document.getElementById('speaker-role');
        const speechText = document.getElementById('speech-text');
 
        if (!speakerAvatar || !speakerName || !speakerRole || !speechText) {
            console.error('[DEBUG] Speech elements not found!');
            resolve();
            return;
        }
 
        speakerAvatar.src = getAvatarForRole(player ? player.role : 'villager');
        speakerName.textContent = data.player;
        speakerRole.textContent = roleInfo ? roleInfo.name : '未知';
 
        speechText.textContent = '';
 
        // 打字机效果
        let i = 0;
        const text = data.content || '';
 
        function typeWriter() {
            if (i < text.length) {
                speechText.textContent += text.charAt(i);
                i++;
                currentTypeWriterTimeout = setTimeout(typeWriter, 50);
            }
        }
        typeWriter();
 
        // 等待一段时间让用户阅读发言
        const displayTime = Math.min(text.length * 50 + 1000, 8000); // 最多显示8秒
        currentSpeechTimeout = setTimeout(() => {
            // 非自动播放模式下保持发言区域显示，让用户点击"下一位"按钮
            // 自动播放模式下不隐藏，让下一条消息覆盖
            unhighlightSpeaker(data.player_id);
 
            console.log('[DEBUG] Speech display complete, autoPlay:', gameState.autoPlay);
 
            // 如果是自动播放模式，检查是否有队列中的发言
            if (gameState.autoPlay && gameState.speechQueue && gameState.speechQueue.length > 0) {
                const nextData = gameState.speechQueue.shift();
                console.log('[DEBUG] Auto playing next queued speech:', nextData.data.player);
                setTimeout(() => onSpeech(nextData.data).then(nextData.resolve), 500);
                resolve();
            } else if (!gameState.autoPlay) {
                // 手动模式：发言显示完毕，暂停消息队列，等待用户点击"下一位"
                console.log('[DEBUG] Manual mode: pausing message queue, waiting for user to click next button');
                console.log('[DEBUG] Current speech queue length:', gameState.speechQueue ? gameState.speechQueue.length : 0);
                // 设置 messageResumeResolve 来暂停整个消息队列
                // 如果已经有 messageResumeResolve，先清理旧的，避免内存泄漏和状态混乱
                if (messageResumeResolve) {
                    console.log('[DEBUG] Clearing stale messageResumeResolve before setting new one');
                    const oldResolve = messageResumeResolve;
                    messageResumeResolve = null;
                    oldResolve();
                }
                messageResumeResolve = resolve;
            } else {
                resolve();
            }
        }, displayTime);
    });
}

        return;

                p.querySelector('div').classList.remove('border-gothic-gold');

                    if (countBadge) {
                        countBadge.textContent = count;
                        countBadge.style.display = 'flex';
                    }
                }
            }
        }
        
        await delay(delayPerVote);
        
        if (targetPlayer) {
            const targetSeat = document.querySelector(`.seat[data-player-id="${targetPlayer.id}"]`);
            if (targetSeat) {
                targetSeat.classList.remove('seat-active');
            }
        }

}

// ==================== MVP 投票 ====================
function showMVPVote(players) {
    // 生成 MVP 候选人
    const candidates = [];
    
    // 狼人胜利时，选择存活的狼人
    if (gameState.gameResult === 'wolf') {
        const aliveWolves = players.filter(p => p.is_alive && p.role === 'werewolf');
        candidates.push(...aliveWolves);
    }
    
    // 好人胜利时，选择存活的神民
    if (gameState.gameResult === 'good') {
        const aliveSpecialRoles = players.filter(p => p.is_alive && ['seer', 'witch', 'hunter', 'guard'].includes(p.role));
        candidates.push(...aliveSpecialRoles);
    }
    
    // 如果候选人不足，添加其他存活玩家
    const alivePlayers = players.filter(p => p.is_alive);
    alivePlayers.forEach(player => {
        if (!candidates.includes(player)) {
            candidates.push(player);
        }
    });
    
    // 限制候选人数量
    const mvpCandidates = candidates.slice(0, 5);
    
    // 渲染 MVP 候选人
    const mvpCandidatesEl = document.getElementById('mvp-candidates');
    mvpCandidatesEl.innerHTML = '';
    
    mvpCandidates.forEach(candidate => {
        const candidateEl = document.createElement('div');
        candidateEl.className = 'mvp-candidate flex flex-col items-center cursor-pointer hover:scale-110 transition-transform';
        candidateEl.dataset.playerId = candidate.id;
        
        candidateEl.innerHTML = `
            <div class="w-16 h-16 rounded-full overflow-hidden border-2 border-gothic-red">
                <img src="${getAvatarForRole(candidate.role)}" alt="${candidate.name}" class="w-full h-full object-cover">
            </div>
            <div class="text-xs font-gothic mt-1 text-gothic-light text-center">${candidate.name}</div>
        `;
        
        candidateEl.addEventListener('click', () => {
            console.log('[DEBUG] MVP candidate clicked:', candidate.name);
            playSound({id: 'button-click'});
            document.querySelectorAll('.mvp-candidate').forEach(c => {
                c.querySelector('div').classList.remove('border-gothic-gold');
            });
            candidateEl.querySelector('div').classList.add('border-gothic-gold');
            gameState.selectedMVP = candidate.id;
            console.log('[DEBUG] Selected MVP:', gameState.selectedMVP);
            document.getElementById('submit-mvp-btn').disabled = false;
        });
        
        mvpCandidatesEl.appendChild(candidateEl);
    });
    
    // 显示 MVP 投票模态框
    toggleMvpVoteModal();
    document.getElementById('submit-mvp-btn').disabled = true;
}

function renderSeats() {
    console.log('[DEBUG] renderSeats called, players:', gameState.players);
    const seatsContainer = document.getElementById('seats-container');
    if (!seatsContainer) {
        console.error('[DEBUG] Seats container not found!');
        return;
    }
 
    seatsContainer.innerHTML = '';
 
    const seatCount = gameState.players.length;
    if (seatCount === 0) {
        console.warn('[DEBUG] No players to render seats');
        return;
    }
 
    const angleStep = (2 * Math.PI) / seatCount;
    // 根据人数动态调整半径，确保不同人数下座位分布均匀且不重叠
    const radius = seatCount <= 6 ? 40 : seatCount <= 8 ? 42 : 44;
 
    gameState.players.forEach((player, index) => {
        console.log('[DEBUG] Rendering seat for player:', player.name, player.role);
        const angle = index * angleStep - Math.PI / 2;
        const x = 50 + radius * Math.cos(angle);
        const y = 50 + radius * Math.sin(angle);
 
        const seat = document.createElement('div');
        seat.className = 'seat absolute w-16 h-28 flex flex-col items-center cursor-pointer';
        seat.style.left = `${x}%`;
        seat.style.top = `${y}%`;
        seat.style.transform = 'translate(-50%, -50%)';
        seat.dataset.playerId = player.id;
 
        const roleInfo = getRoleInfo(player.role);
        const isDead = !player.is_alive;
 
        if (isDead) {
            seat.classList.add('seat-dead');
        }
 
        seat.innerHTML = `
            <div class="avatar-wrapper w-12 h-12 rounded-full overflow-hidden border-2 border-gothic-red ${isDead ? 'grayscale brightness-50' : ''}">
                <img src="${getAvatarForRole(player.role)}" alt="${player.name}" class="w-full h-full object-cover">
            </div>
            <div class="player-name text-xs font-gothic mt-1 text-gothic-light text-center">${player.name}</div>
            <div class="player-role-label">${roleInfo ? roleInfo.name : '平民'}</div>
        `;
 
        seatsContainer.appendChild(seat);
    });
 
    console.log('[DEBUG] Seats rendered successfully');
}

            skillType = 'seer';
            nightClass = 'night-seer-active';
            targetName = extractTargetFromAction(a.action);
        } else if (actionLower.includes('投毒') || actionLower.includes('毒杀')) {
            skillType = 'witch-poison';
            nightClass = 'night-witch-active';
            targetName = extractTargetFromAction(a.action);
        } else if (actionLower.includes('救')) {
            skillType = 'witch-heal';
            nightClass = 'night-witch-active';
            targetName = extractTargetFromAction(a.action);
        } else if (actionLower.includes('保护')) {
            skillType = 'guard';
            nightClass = 'night-guard-active';
            targetName = extractTargetFromAction(a.action);
        }
 
        if (skillType && targetName) {
            const targetPlayer = gameState.players.find(p => p.name === targetName);
            if (targetPlayer) {
                const actorSeat = document.querySelector(`.seat[data-player-id="${playerObj.id}"]`);
                const targetSeat = document.querySelector(`.seat[data-player-id="${targetPlayer.id}"]`);
 
                // 添加角色特定高亮
                if (actorSeat && nightClass) actorSeat.classList.add(nightClass);
                if (targetSeat) targetSeat.classList.add('seat-active');
 
                // 显示连线动画
                showActionLine(actorSeat, targetSeat, skillType, actionDelay);
 
                // 显示行动文字提示
                showActionTooltip(actorSeat, a.action);
 
                setTimeout(() => {

                }, actionDelay);
            }

                showActionTooltip(actorSeat, a.action);

                }, actionDelay);
            }
        }
 
        await delay(actionDelay);
    }
 
    // 退出夜晚模式
    if (seatsContainer) seatsContainer.classList.remove('night-mode');
}
 
function extractTargetFromAction(actionText) {
    const patterns = [
        /击杀\s*(.+)/, /杀害\s*(.+)/, /选择击杀\s*(.+)/,
        /查验了\s*(.+)/, /查验\s*(.+)/,
        /对\s*(.+?)\s*使用/, /投毒\s*(.+)/, /毒杀\s*(.+)/,
        /守护了\s*(.+)/, /守护\s*(.+)/,
        /救了\s*(.+)/, /保护了\s*(.+)/
    ];
    for (const pattern of patterns) {
        const match = actionText.match(pattern);
        if (match) {
            let name = match[1].trim();
            name = name.replace(/[：:，。！？、；:]/g, '');
            name = name.replace(/\s*(是狼人|是好人|狼人|好人).*$/g, '');
            return name.trim();
        }
    }
    return null;
}
 
function showActionLine(fromSeat, toSeat, lineClass = "", duration = 2500) {

// 使用 Web Audio API 生成简单音效
const audioContext = new (window.AudioContext || window.webkitAudioContext)();
 
function playTone(frequency, duration, type = 'sine') {
    try {
        const oscillator = audioContext.createOscillator();
        const gainNode = audioContext.createGain();
        
        oscillator.connect(gainNode);
        gainNode.connect(audioContext.destination);
        
        oscillator.frequency.value = frequency;
        oscillator.type = type;
        
        gainNode.gain.setValueAtTime(0.3, audioContext.currentTime);
        gainNode.gain.exponentialRampToValueAtTime(0.01, audioContext.currentTime + duration);
        
        oscillator.start(audioContext.currentTime);
        oscillator.stop(audioContext.currentTime + duration);
        
        console.log('[DEBUG] Tone played:', frequency, 'Hz');
    } catch (e) {
        console.log('[DEBUG] Web Audio API error:', e);
    }
}

function shuffleArray(array) {
    const newArray = [...array];
    for (let i = newArray.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [newArray[i], newArray[j]] = [newArray[j], newArray[i]];
    }
    return newArray;
}

function nextSpeech() {
    console.log('[DEBUG] nextSpeech called');
    console.log('[DEBUG] Speech queue length:', gameState.speechQueue ? gameState.speechQueue.length : 0);
    console.log('[DEBUG] Message queue length:', messageQueue.length);
    console.log('[DEBUG] Current phase:', gameState.currentPhase);

    console.log('[DEBUG] WebSocket state:', ws ? ws.readyState : 'closed');
 
    // 清除当前发言的打字机和显示定时器，防止动画未完成时状态混乱
    if (currentTypeWriterTimeout) {
        clearTimeout(currentTypeWriterTimeout);
        currentTypeWriterTimeout = null;
    }
    if (currentSpeechTimeout) {
        clearTimeout(currentSpeechTimeout);
        currentSpeechTimeout = null;
    }
 
    // 先隐藏当前发言区域
    const speechArea = document.getElementById('speech-area');
    if (speechArea) {
        speechArea.classList.add('hidden');
    }
 
    // 如果有 messageResumeResolve，说明消息队列被暂停，先恢复它
    if (messageResumeResolve) {
        console.log('[DEBUG] Resuming message queue');
        const resolve = messageResumeResolve;
        messageResumeResolve = null;
        resolve();
    }
 
    // 检查是否有队列中的发言
    if (gameState.speechQueue && gameState.speechQueue.length > 0) {
        // 从队列中取出下一条发言并显示
        // speechQueue 中存储的是 { data, resolve } 对象
        const nextItem = gameState.speechQueue.shift();
        console.log('[DEBUG] Playing queued speech:', nextItem.data.player);
        // 使用 setTimeout 确保 UI 更新后再显示下一条
        setTimeout(() => {
            onSpeech(nextItem.data).then(nextItem.resolve);
        }, 100);
        return;
    }
 
    console.log('[DEBUG] No queued speech, checking if all speeches complete');
 
    // 如果消息队列中有积压的非发言消息（如 vote_start），先处理它们
    if (messageQueue.length > 0) {
        console.log('[DEBUG] Message queue has', messageQueue.length, 'pending messages, processing next');
        // 恢复消息队列处理
        if (!isProcessingMessage) {
            processNextMessage();
        }
        return;
    }
 
    // 如果队列为空且游戏处于发言阶段，通知后端继续游戏
    // 后端 phase 值为: lobby, night, dawn, day 等
    // 扩展支持更多可能的发言阶段值
    const speechPhases = ['day', 'dawn', 'day_discussion', 'speech'];
    if (speechPhases.includes(gameState.currentPhase)) {
        console.log('[DEBUG] All speeches completed, sending ready_for_next to backend');
 
        // 检查 WebSocket 状态
        if (ws && ws.readyState === WebSocket.OPEN) {
            ws.send(JSON.stringify({ type: 'ready_for_next' }));
        } else {
            console.error('[DEBUG] WebSocket not open, cannot send ready_for_next');
 
            // 尝试重新连接
            if (!ws || ws.readyState === WebSocket.CLOSED || ws.readyState === WebSocket.CLOSING) {
                console.log('[DEBUG] Attempting to reconnect WebSocket');
                connectWS();
            }
        }
    }
}
